    text = text.lower()

    danger_words = [
        "delete",
        "remove",
        "os.remove",
        "shutil.rmtree",
        "subprocess",
        "powershell",
        "cmd",
        "payment",
        "purchase",
        "password",
        "api key",
        "token",
        "send to server",
        "overwrite"
    ]

    for word in danger_words:
        if word in text:
            return True

    # 追加: 危険なファイル操作を検知する
    if "file" in text and ("delete" in text or "remove" in text or "os.remove" in text or "shutil.rmtree" in text):

    # 追加: 危険なコマンドを検知する
    if "subprocess" in text or "powershell" in text or "cmd" in text:
        return True

    # 追加: 危険な送信を検知する
    if "send to server" in text:
        return True

    return False
